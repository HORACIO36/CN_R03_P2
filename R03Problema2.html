Para la superficie de control

Fis = addOutput(Fis,[-1 1],'Name','U');
Fis = addMF(Fis,'U','constant',-1,'Name','NB');
Fis = addMF(Fis,'U','constant',-0.5,'Name','NM');
Fis = addMF(Fis,'U','constant',0,'Name','Z');
Fis = addMF(Fis,'U','constant',0.5,'Name','PM');
Fis = addMF(Fis,'U','constant',1,'Name','PB');
reglas = ["E==N & delE==N => U=NB";
"E==Z & delE==N => U=NM";
"E==P & delE==N => U=Z";
"E==N & delE==Z => U=NM";
"E==Z & delE==Z => U=Z";
"E==P & delE==Z => U=PM";
"E==N & delE==P => U=Z";
"E==Z & delE==P => U=PM";
"E==P & delE==P => U=PB"];
Fis = addRule(Fis,reglas);
figure
gensurf(Fis), title('Superficie de control FIS tipo-1')
